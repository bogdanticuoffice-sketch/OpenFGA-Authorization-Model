# ABAC example tuples
# Format: object#relation@user
#
# NOTE: Condition-based relations (reader with is_within_business_hours,
# dept_reader with same_department) require contextual data at check time.
# When writing those tuples via the API, include the condition name:
#
#   {
#     "user": "user:bob",
#     "relation": "reader",
#     "object": "resource:api-key-service",
#     "condition": { "name": "is_within_business_hours", "context": {} }
#   }
#
# The tuples below show direct (unconditional) assignments.

organization:acme#admin@user:alice
organization:acme#member@user:bob
organization:acme#member@user:charlie

department:engineering#organization@organization:acme
department:engineering#manager@user:alice
department:engineering#member@user:bob
department:engineering#member@user:charlie

department:finance#organization@organization:acme
department:finance#manager@user:dave
department:finance#member@user:eve

resource:api-key-service#organization@organization:acme
resource:api-key-service#department@department:engineering
resource:api-key-service#owner@user:alice
# bob can read api-key-service only during business hours (conditional tuple)
# resource:api-key-service#reader@user:bob  (with condition: is_within_business_hours)

resource:budget-report#organization@organization:acme
resource:budget-report#department@department:finance
resource:budget-report#owner@user:dave
# eve can read budget-report only when user_dept == "finance" (conditional tuple)
# resource:budget-report#dept_reader@user:eve  (with condition: same_department)
