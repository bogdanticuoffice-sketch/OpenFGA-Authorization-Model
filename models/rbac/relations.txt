# RBAC example tuples
# Format: object#relation@user

organization:acme#admin@user:alice
organization:acme#member@user:bob
organization:acme#member@user:charlie
organization:acme#viewer@user:david

team:platform#organization@organization:acme
team:platform#owner@user:alice
team:platform#member@user:bob
team:platform#viewer@user:charlie

# alice is owner of team:platform → inherits editor → inherited by project via editor from team
# bob is member of team:platform → inherits viewer on project via viewer from team
project:backend-api#team@team:platform
project:backend-api#owner@user:alice
project:backend-api#editor@user:bob

resource:user-service#project@project:backend-api
resource:user-service#owner@user:alice
resource:user-service#editor@user:bob
resource:user-service#viewer@user:charlie

document:user-service-spec#resource@resource:user-service
document:user-service-spec#owner@user:alice
document:user-service-spec#editor@user:bob
document:user-service-spec#viewer@user:charlie
